package org.jahia.utils.pagination;


import java.util.HashMap;
import java.util.List;
import java.util.Map;

/**
 * Sometimes you already have a list and you want to paginate it.
 * That's what this is for. you don't get searching or sorting without it.
 *
 * @author MJ
 */
public class PaginatablePrebuiltList extends PaginatableList
{
    public static final Map filterableFields = new HashMap();


    /**
     * A basic constructor.
     */
    public PaginatablePrebuiltList(Integer itemsPerPage, List list)
    {
        clear();
        addAll(list);
        this.itemsPerPage = itemsPerPage.intValue();
    }

    public String toString()
    {
        StringBuffer d = new StringBuffer();
        d.append("PaginatablePrebuiltList containing ").append(size())
            .append(" elements.");
        return d.toString();
    }
}
